#include <stdio.h>
char str[1000];

int main(){
	while (true) {
		/*
		* char ch;
		* scanf("%[^\n]%c", str, &ch);
		* printf("%s\n", str);
		*/
		scanf("%[^\n]%*c", str); //与上面等价。
		/*
		* 1. %[] 读取一个字符集。
		* 2. %n[^\n] 读取最多n个字符(n省略，意为读取任意多的字符)， 或者遇到'\n'提前终止读取(\n和替换为其他任意字符)。
		* 3. %n[a-z] n同上， 或者遇到非a到z的字符停止。 
		* 4. %n[0-9] 同上
		* 5. %n[^0-9] 同上
		* 6. %n[a-zA-Z] 同上
		* 7. '-'是范围连接符，当然也可以直接列出需要读入的字符
		* 8. %*c 意为跳过下一个字符（本例中，跳过最后的换行符）
		*/
	}
	return 0;
}
